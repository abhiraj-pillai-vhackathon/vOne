package com.example.app;

import androidx.appcompat.app.AppCompatActivity;

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;

public class confirmtionPage extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_confirmtion_page);

        Button backToCanteen = findViewById(R.id.backToCanteen);

        backToCanteen.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                Intent backToCanteen = new Intent(confirmtionPage.this,canteen.class);
                startActivity(backToCanteen);
                finish();

            }
        });



    }
}
